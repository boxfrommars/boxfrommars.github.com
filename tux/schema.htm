Схемы API

Схемы для ответов на GET запросы, написаны в формате json-schema draft v4

Заказ

{
    "id": "/order",
    "title": "Order Schema",
    "type": "object",
    "properties": {
        "id": {
            "type": "integer"
        },
        "customer": {
            "$ref": "/customer"
        },
        "areaId": {
            "type": "integer"
        },
        "rides": {
            "type": "array",
            "items": {
                "$ref": "/ride"
            }
        }
    },
    "required": ["id", "customer", "areaId", "rides"]
}

Поездка

    "id": "/ride",
    "title": "Ride Schema",
    "type": "object",
    "properties": {
        "id": {
            "type": "integer"
        },
        "orderType": {
            "type": "string",
            "enum": ["emergency", "normal", "preorder"]
        },
        "preOrderTime": {
            "oneOf": [{"type": "null"}, {"$ref": "/time"}]
        },
        "alertTypeId": {
            "type": "integer"
        },
        "orderedProperties": {
            "type": "object",
            "properties": {
                "options": {
                    "type": "array",
                    "items": {
                        "type": "integer"
                    }
                },
                "type": {
                    "type": "object",
                    "properties": {
                        "id": {
                            "type": "integer"
                        },
                        "classId": {
                            "oneOf": [{"type": "null"}, {"type": "integer"}]
                        }
                    },
                    "required": ["id"]
                }
            }
        },
        "phone": {
            "type": "string",
            "pattern": "\\d+"
        },
        "description": {
            "type": "string"
        },
        "departureAddress": {
            "oneOf": [{"type": "null"}, {"$ref": "/address"}]
        },
        "destinationAddress": {
            "$ref": "/address"
        },
        "vehicle": {
            "$ref": "/vehicle"
        },
        "timeline": {
            "$ref": "/timeline"
        },
        "path": {
            "type": "array",
            "items": {
                "type": "object",
                "properties": {
                    "point": {
                        "$ref": "/geopoint"
                    },
                    "time": {
                        "$ref": "/time"
                    }
                },
                "required": ["point", "time"]
            }
        },
        "costDetails": {
            "type": "object",
            "properties": {
                "callCost": {
                    "type": "numeric",
                    "minimum": 0
                },

                    "type": "numeric",
                    "minimum": 0
                },
                "waitingTime": {
                    "type": "numeric",
                    "minimum": 0
                },
                "pathCost": {
                    "type": "numeric",
                    "minimum": 0
                },

                    "type": "numeric",
                    "minimum": 0
                },
                "cost": {
                    "type": "numeric",
                    "minimum": 0
                }
            },
            "required": []
        }
    },

    "required": ["id", "orderType", "preOrderTime", "phone", "timeline"]
}
