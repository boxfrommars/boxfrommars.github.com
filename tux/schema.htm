Схемы API

Заказ

{
    "id": "/order",
    "title": "Order Schema",
    "type": "object",
    "properties": {
        "id": {
            "type": "integer"
        },
        "customer": {
            "$ref": "/customer"
        },
        "areaId": {
            "type": "integer"
        },
        "rides": {
            "type": "array",
            "items": {
                "$ref": "/ride"
            }
        }
    },
    "required": ["id", "customer", "areaId", "rides"]
}

            "type": "string",
